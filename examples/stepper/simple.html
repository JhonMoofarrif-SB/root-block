const stepContents = {
  1: '<h3>Paso 1</h3><input placeholder="Nombre" />',
  2: '<h3>Paso 2</h3><input placeholder="Email" />',
  3: '<h3>Paso 3</h3><select>...</select>',

// 2Ô∏è‚É£ Funci√≥n para renderizar el paso actual
function renderStep() {
  // Actualizar contenido din√°mico
  document.getElementById('stepContent').innerHTML = stepContents[currentStep];

  // Actualizar stepper visual
  updateStepperVisual();

  // Actualizar botones
  document.getElementById('btnPrev').disabled = (currentStep === 1);
  document.getElementById('btnNext').textContent =
    (currentStep === totalSteps) ? 'Finalizar ‚úì' : 'Siguiente ‚Üí';
}

// 3Ô∏è‚É£ Actualizar estados del stepper
function updateStepperVisual() {
  const steps = document.querySelectorAll('.sb-ui-stepper__step');

  steps.forEach((step, index) => {
    const stepNumber = index + 1;
    const circle = step.querySelector('.sb-ui-stepper__step-circle');

    // Remover clases anteriores
    step.classList.remove(
      'sb-ui-stepper__step--empty-default',
      'sb-ui-stepper__step--empty-active',
      'sb-ui-stepper__step--filled-default'
    );

    if (stepNumber < currentStep) {
      // ‚úÖ Completado: verde con check
      step.classList.add('sb-ui-stepper__step--filled-default');
      circle.innerHTML = '<svg>...check icon...</svg>';
    } else if (stepNumber === currentStep) {
      // üîµ Activo: amarillo con n√∫mero
      step.classList.add('sb-ui-stepper__step--empty-active');
      circle.textContent = stepNumber;
    } else {
      // ‚ö™ Pendiente: blanco con n√∫mero gris
      step.classList.add('sb-ui-stepper__step--empty-default');
      circle.textContent = stepNumber;
    }
  });
}

// 4Ô∏è‚É£ Navegaci√≥n simple
function next() {
  if (currentStep < totalSteps) {
    currentStep++;
    renderStep();
  } else {
    alert('¬°Formulario completado! üéâ');
  }
}

function prev() {
  if (currentStep > 1) {
    currentStep--;
    renderStep();
  }
}
